export PATH=~/.npm-global/bin:$PATH
source ~/.profile
npm config set prefix /usr/local
npm install -g pm2
sudo chown -R $USER /usr/local

sudo apt-get update

sudo apt-key fingerprint 0EBFCD88

sudo apt-get update
